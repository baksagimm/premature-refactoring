Objective A: Print "Hello Kyubi" x2 times

            function executeA() {
                // procedual coding

                msg1.textContent = 'Hello Kyubi!';
                print_el_a.append(msg1);
        
                let msg2 = document.createElement('div');
                msg2.textContent = 'Hello Kyubi!';
                print_el_a.append(msg2);
            }
